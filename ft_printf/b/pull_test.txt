#!/bin/bash
sleep_interval=1m
echo sleep interval = $sleep_interval
have_new_code() 
{
	echo exec: git pull
	pull_result=$(git pull)
	echo $pull_result
	if [ "$pull_result" == "Already up to date." ]
	then
		return 0
	fi
	if [ "$pull_result" ==  "Уже обновлено." ]
	then
		return 0
	fi
	
	return 1
}

build()
{
	cmake ..
	echo cmake .. result = $?
	if (($? != 0))
	then
		exit $?
	fi
	cmake --build .
	echo cmake --build . result = $?
	if (($? != 0))
	then
		exit $?
	fi
}

run()
{
	echo running tests
	./ft_printf_tests/ft_printf_tests 2>err.txt 1>out.txt 
}

email()
{
	echo sending email
	mail -s tests forspam28@ya.ru <err.txt 
	mail -s tests forspam28@ya.ru <out.txt 
}

handle_new_code()
{
	have_new_code
	if [ $? == 1 ]
	then
		echo "new code pulled"
	else
		echo "no new"
		return 0
	fi
	build
	run
	email

}

loop_body()
{
	handle_new_code
	sleep $sleep_interval
}



echo "Press [CTRL+C] to stop.."
while true
do
	loop_body
done
